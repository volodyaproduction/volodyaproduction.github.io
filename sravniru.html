def all_clf(X_train, y_train, X_test, y_test):
    
#   1.  
    lr = LogisticRegression(penalty = 'l2',
                            max_iter = 30000)
    lr.fit(X_train, y_train)
    
#   2. RandomForest
    clf_rf = RandomForestClassifier()
    parameters_rf = {'min_samples_leaf': [5, 10, 15],
                     'max_features': [3],
                     'n_estimators': [100],
                     'max_depth': [5, 7, 10, 12, 15]}

    grid_search_cv_clf_rf = RandomizedSearchCV(clf_rf, parameters_rf, cv = 5, n_jobs = -1)
    grid_search_cv_clf_rf.fit(X_train, y_train)
    best_clf_rf = grid_search_cv_clf_rf.best_estimator_
    
#   3. GradientBoosting
    clf_gb = GradientBoostingClassifier()
    parameters_gb = { 'min_samples_leaf': [5, 10, 15],
                      'max_features': [3],
                      'n_estimators': [100],
                      'max_depth': [3, 5, 7, 10, 12]}  
    grid_search_cv_clf_gb = RandomizedSearchCV(clf_gb, parameters_gb, cv = 5, n_jobs = -1)
    grid_search_cv_clf_gb.fit(X_train, y_train)
    best_clf_gb = grid_search_cv_clf_gb.best_estimator_
    
#   4. XGB-classifier
    clf_xgb = xgboost.XGBClassifier(use_label_encoder = False, eval_metric = 'mlogloss')
    parameters_xgb = {"learning_rate"    : [0.05, 0.10, 0.15, 0.20, 0.25, 0.30 ],
                      "max_depth"        : [ 3, 4, 5, 6, 8, 10, 12, 15],
                      "min_child_weight" : [ 1, 3, 5, 7 ],
                      "gamma"            : [ 0.0, 0.1, 0.2 , 0.3, 0.4 ],
                      "colsample_bytree" : [ 0.3, 0.4, 0.5 , 0.7 ] }
    grid_search_cv_clf_xgb = RandomizedSearchCV(clf_xgb, parameters_xgb, cv = 5, n_jobs = -1)
    grid_search_cv_clf_xgb.fit(X_train, y_train)
    best_clf_xgb = grid_search_cv_clf_xgb.best_estimator_

#   5. K-neighbour classifier
    clf_knn = KNeighborsClassifier()
    parameters_knn = {'n_neighbors': np.arange(1, 50, 1),
                      'weights': ['uniform', 'distance']}
    grid_search_cv_clf_knn = RandomizedSearchCV(clf_knn, parameters_knn, cv = 5, n_jobs = -1)
    grid_search_cv_clf_knn.fit(X_train, y_train)
    best_clf_knn = grid_search_cv_clf_knn.best_estimator_
    
#   6. Neural network
    clf_nn = MLPClassifier()
    parameters_nn = {'hidden_layer_sizes': [5, 10, 15, 20, 25, 30, 35, 40]}
    grid_search_cv_clf_nn = GridSearchCV(clf_nn, parameters_nn, cv = 5)
    grid_search_cv_clf_nn.fit(X_train, y_train)
    best_clf_nn = grid_search_cv_clf_nn.best_estimator_
    
    d = {'Accuracy': [accuracy_score(y_test, lr.predict(X_test)), 
                          accuracy_score(y_test, best_clf_rf.predict(X_test)),
                          accuracy_score(y_test, best_clf_gb.predict(X_test)),
                          accuracy_score(y_test, best_clf_xgb.predict(X_test)),
                          accuracy_score(y_test, best_clf_knn.predict(X_test)),
                          accuracy_score(y_test, best_clf_nn.predict(X_test))],
         'Sensitivity (TPR)': [recall_score(y_test, lr.predict(X_test)), 
                               recall_score(y_test, best_clf_rf.predict(X_test)),
                               recall_score(y_test, best_clf_gb.predict(X_test)),
                               recall_score(y_test, best_clf_xgb.predict(X_test)),
                               recall_score(y_test, best_clf_knn.predict(X_test)),
                               recall_score(y_test, best_clf_nn.predict(X_test))],
         'Specifity (TNR)': [recall_score(y_test, lr.predict(X_test), pos_label = 0), 
                             recall_score(y_test, best_clf_rf.predict(X_test), pos_label = 0),
                             recall_score(y_test, best_clf_gb.predict(X_test), pos_label = 0),
                             recall_score(y_test, best_clf_xgb.predict(X_test), pos_label = 0),
                             recall_score(y_test, best_clf_knn.predict(X_test), pos_label = 0),
                             recall_score(y_test, best_clf_nn.predict(X_test), pos_label = 0)],
        'Precision': [precision_score(y_test, lr.predict(X_test)), 
                          precision_score(y_test, best_clf_rf.predict(X_test)),
                          precision_score(y_test, best_clf_gb.predict(X_test)),
                          precision_score(y_test, best_clf_xgb.predict(X_test)),
                          precision_score(y_test, best_clf_knn.predict(X_test)),
                          precision_score(y_test, best_clf_nn.predict(X_test))],
        'F1_score':  [f1_score(y_test, lr.predict(X_test)), 
                          f1_score(y_test, best_clf_rf.predict(X_test)),
                          f1_score(y_test, best_clf_gb.predict(X_test)),
                          f1_score(y_test, best_clf_xgb.predict(X_test)),
                          f1_score(y_test, best_clf_knn.predict(X_test)),
                          f1_score(y_test, best_clf_nn.predict(X_test))],
        'ROC AUC': [roc_auc_score(y_test, lr.predict_proba(X_test)[:, 1]), 
                          roc_auc_score(y_test, best_clf_rf.predict_proba(X_test)[:, 1]),
                          roc_auc_score(y_test, best_clf_gb.predict_proba(X_test)[:, 1]),
                          roc_auc_score(y_test, best_clf_xgb.predict_proba(X_test)[:, 1]),
                          roc_auc_score(y_test, best_clf_knn.predict_proba(X_test)[:, 1]),
                          roc_auc_score(y_test, best_clf_nn.predict_proba(X_test)[:, 1])]}
    return pd.DataFrame(data = d, index = ['Logistic Regression', 'Random Forest', 'Gradient Boosting', 'XGB', 'KNN', 'Neural network']);
